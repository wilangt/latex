\newcommand\barre{\noindent\paragraph{}\hrulefill}

\newcommand{\includexo}[1]{
	\def\temp{#1}\ifx\temp\empty
  	
	\else
		\input{body/exos/#1}
	\fi
}

\newcommand{\header}[1]{
	\ifnum\hidden=0
	\newpage
	\section{#1}
	\else
	\fi
}

\newcommand{\eleve}[4]{
	\ifnum\hidden=0
	
	\begin{minipage}{\textwidth}
	\barre
	
	\begin{center}\subsection*{Colle d'informatique - #1}\end{center}
	\setcounter{exercice}{0}
	\includexo{#2}
	\includexo{#3}
	\includexo{#4}
	\paragraph{} Note :
	\end{minipage}

	\else
	\fi
}

\newcommand{\bonus}[4]{
	\ifnum\hidden=0
		
	\barre

	\begin{center}\subsection*{Exercice Bonus - #1}\end{center}
	\setcounter{exercice}{-1}
	\includexo{#2}
	\setcounter{exercice}{-1}
	\includexo{#3}
	\setcounter{exercice}{-1}
	\includexo{#4}

	\else
	\fi
}

\newcommand{\colle}[9]{
	\ifnum\hidden=0
	
	\header{#1}

	\eleve{#1}{#2}{#3}{}

	\eleve{#1}{#4}{#5}{}

	\eleve{#1}{#6}{#7}{#8}
	
	\def\temp{#9}\ifx\temp\empty
  	
	\else
		\bonus{#1}{#9}{}{}
	\fi

	\else
	\fi
}
